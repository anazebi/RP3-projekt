CREATE TABLE Rooms(roomID int NOT NULL IDENTITY PRIMARY KEY, roomNo varchar(50) UNIQUE, roomType varchar(50) NOT NULL, beds varchar(50) NOT NULL, price bigint NOT NULL, booked varchar(50));
CREATE TABLE Guests(guestID int NOT NULL IDENTITY PRIMARY KEY, guestName varchar(50) NOT NULL, gender varchar(50) NOT NULL, idNumber varchar(50) NOT NULL, dateIn varchar(50) NOT NULL, dateOut varchar(50) NOT NULL, checkedIn varchar(50), checkedOut varchar(50), roomID int FOREIGN KEY REFERENCES Rooms(roomID));
CREATE TABLE Reservations(reservationID int NOT NULL IDENTITY PRIMARY KEY, guestID int FOREIGN KEY REFERENCES Guests(guestID), guestName varchar(50) NOT NULL, roomID int FOREIGN KEY REFERENCES Rooms(roomID), roomType varchar(50) NOT NULL, dateIn varchar(50) NOT NULL, dateOut varchar(50) NOT NULL, nights int NOT NULL, price bigint NOT NULL, special bigint, overall bigint NOT NULL, remarks varchar(400), paid bigint);
